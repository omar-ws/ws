THRESHOLDS = {
    "temperature":   {"warn": 45,  "critical": 60},
    "voltage_delta": {"warn": 50,  "critical": 100},
    "charge_rate":   {"warn": 1.5, "critical": 2.0},
    "resistance":    {"warn": 80,  "critical": 120},
}

def classify_battery_state(telemetry):
    if telemetry["temperature"] > THRESHOLDS["temperature"]["critical"]:
        return "CRITICAL"
    
    if telemetry['voltage_delta'] > THRESHOLDS['voltage_delta']['critical']:
        return 'CRITICAL'   
    
    if telemetry['charge_rate'] > THRESHOLDS['charge_rate']['critical']:
        return 'CRITICAL'    
    
    if telemetry['resistance'] > THRESHOLDS['resistance']['critical']:
        return 'CRITICAL'    
    
    
    if telemetry["temperature"] > THRESHOLDS["temperature"]["warn"]:
        return "WARN"

    if telemetry["voltage_delta"] > THRESHOLDS["voltage_delta"]['warn']:
        return 'WARN'

    if telemetry["charge_rate"] > THRESHOLDS["charge_rate"]['warn']:
        return 'WARN'

    if telemetry["resistance"] > THRESHOLDS["resistance"]['warn']:
        return 'WARN'
    return "NORMAL"
